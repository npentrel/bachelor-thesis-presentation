\begin{frame}
  \frametitle{Primitives}
  \begin{module}[id=primitives]
		\\
		\noindent
		\begin{definition}
		  \defi{Primitives} are the different elements of representation that can be divided into structural and semantic primitives.
		\end{definition}

		\noindent
		Visual and normal tree-like graph language have different primitives and can be mapped to each other easily.\\
		\\
		\noindent
		$\rightarrow$ information graphs can be built on these primitives.

  \end{module}
\end{frame}
